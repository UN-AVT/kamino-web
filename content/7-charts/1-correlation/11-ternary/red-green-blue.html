Plot

position, shape, color, size

theme_opts <- theme(
  text = element_text(family = "inconsolata"),
  legend.position = "none",
  legend.title = element_blank(),
  tern.axis.title.L = element_text(hjust = 0.5),
  tern.axis.title.R = element_text(hjust = 0.5),
  panel.background = element_blank(),
  panel.border = element_blank(),
  plot.background = element_blank(),
  plot.margin = unit(c(0.25, 0.25, 0.25, 0.25), "in")
)

v1 <- ggtern(data=df, aes(b, r, g)) + 
  geom_point(aes(color = hex), size = 2, alpha=1.0) + 
  scale_color_identity() +
  coord_tern() +
  theme_minimal() +
  theme_nomask() +
  theme_opts +
  labs(title = "Named Colors", 
           subtitle = "by RGB values") +
           xlab("Blue") + 
           ylab("Red") +
           zlab("Green")

# v1

girafe(ggobj = v1, width_svg = 1280/72, height_svg = 720/72,
       options = list(opts_sizing(rescale = TRUE, width = 1.0))
)
