Load Data

df1 <- read_excel('archetypes/ghg-emissions-by-country/ghg-emissions-by-country.xlsx', sheet=1)

df1

GHG emissions (million tons CO2 per year)

v1 <- ggplot(df1, aes(x=`Consuming countries`, y=`Producing countries`, fill=`GHG emissions`))+
      geom_tile()+
      scale_fill_distiller(palette = "Oranges") +
      labs(x = "Consuming Countries", y = "Producing Countries", 
       title = "GHG emissions (million tons CO2 per year)",
       subtitle = "")


girafe(ggobj = v1, width_svg = 1280/72, height_svg = 720/72,
      options = list(opts_sizing(rescale = TRUE, width = 1.0)))

Save plot

as vector or raster image

# full version
ggsave(filename = "assets/ghg-emissions-heatmap.svg", plot = v1, width=1280/72, height=720/72)
