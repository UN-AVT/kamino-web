df

Wrangle

filter, from long to wide, enrich

df_wrangle <- filter(df, TIME == 2020)
df_wrangle <- filter(df_wrangle, SUBJECT %in% c("PRY_START", "PRY15YREXP", "PRY_TOP"))
df_wrangle <- filter(df_wrangle, MEASURE == "USD")
df_wrangle <- spread(df_wrangle, SUBJECT, Value)
df_wrangle <- df_wrangle %>% select(LOCATION, PRY_START, PRY15YREXP, PRY_TOP)
df_wrangle$COUNTRY <- countrycode(df_wrangle$LOCATION, origin = 'iso3c', destination = 'country.name' )
df_wrangle$REGION <- countrycode(df_wrangle$LOCATION, origin = 'iso3c', destination = 'region' )
df_wrangle

    plot.title = element_text(color = "black", size = 14, face = "bold"),

    axis.title.x = element_blank(),

    panel.border = element_blank(),

    panel.grid.minor = element_blank(), # remove minor gridlines
    panel.grid.major = element_line(size = 0.5, linetype = 'dotted', colour = "#BDAC81"),

shape_map <- c("PRY_START" = 21, "PRY15YREXP" = 5, "PRY_TOP" = 4)

v1 <- ggplot(df_wrangle) +  
  geom_segment( aes(x = PRY_START, xend = PRY_TOP, y = COUNTRY, yend = COUNTRY), lineend = "butt", linejoin = "round", size=1.0, color="#293835" ) +
  geom_point( aes(x = PRY_START, y = COUNTRY, shape = "PRY_START" ), size = 5, fill = "#B50810", color = "#B50810", alpha = 0.9) +
  geom_point( aes(x = PRY15YREXP, y = COUNTRY, shape = "PRY15YREXP"), size = 3, color = "#293835", stroke = 1.5, alpha = 1.0) +
  geom_point( aes(x = PRY_TOP, y = COUNTRY, shape = "PRY_TOP"), size = 3, color = "#293835", stroke = 1.5, alpha = 1.0) +
  scale_x_continuous(labels=scales::dollar_format(), sec.axis = dup_axis()) +
  scale_shape_manual(values = shape_map, labels = c("START", "15YR", "TOP")) +
  coord_cartesian(clip = "off") +
  labs(title = "TEACHERS' SALARIES",
       subtitle = "Upper Secondary, starting, 15 yr, and top pay (USD)",
       caption = "Source: OECD",
       x = NULL,
       y = NULL) +
  theme_minimal() + 
  theme_opts

girafe(ggobj = v1, width_svg = 1280/72, height_svg = 1080/72,
       options = list(opts_sizing(rescale = FALSE, width = 0.95)))
