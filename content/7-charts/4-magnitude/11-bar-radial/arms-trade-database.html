A radial bar chart is a bar chart plotted in polar coordinates. Because the bars are plotted on a polar axis with different radii, the lengths can not be compared as precisely as other methods.

Ingest the data

series, values, and dates

df_suppliers = read.csv("archetypes/arms-trade-database/TIV-Export-Top-10-2019-suppliers.csv", header = TRUE, stringsAsFactors = FALSE)
df_suppliers
df_recipients = read.csv("archetypes/arms-trade-database/TIV-Export-Top-10-2019-recipients.csv", header = TRUE, stringsAsFactors = FALSE)
df_recipients

Wrangle the data

Clean, arrange, tail, and factor

df_suppliers <- df_suppliers %>%
  filter(!is.na(X2019)) %>%
  arrange(X2019) %>%
  tail(6) %>%
  mutate(Supplier=factor(Supplier, Supplier))
df_suppliers
df_recipients <- df_recipients %>%
  filter(!is.na(X2019)) %>%
  arrange(X2019) %>%
  tail(6) %>%
  mutate(Recipient=factor(Recipient, Recipient))
df_recipients

Plot

Suppliers

theme_opts <- theme(
    text = element_text(family = "inconsolata", size = 20), 
    plot.title = element_text(color = "black", size = 12, face = "bold"),
    plot.subtitle = element_text(color = "black", size = 8),
    plot.caption = element_text(color = "#555555", size = 8),
    axis.title.x = element_blank(),
    axis.title.y = element_blank(),
    axis.text = element_blank(),
    panel.border = element_blank(),
    panel.background = element_blank(),
    panel.grid.minor = element_blank(), # remove minor gridlines
    panel.grid.major.x = element_blank(), # remove x (vertical) gridlines
    legend.title = element_blank(), # remove legend title
    legend.position='none'
  )

palette <- c( "#e53935", "#FB8C00", "#039BE5", "#00ACC1", "#00897B", "#7CB342" )

v1 = ggplot(data=df_suppliers, aes(x=Supplier, y=X2019) ) +
  geom_bar(aes(fill=Supplier), stat="identity") +
  geom_text(aes( y = 0, label = paste(Supplier," ")), hjust = 1, size = 6, family = "inconsolata") +
  scale_fill_manual(values=rev(palette)) +
  xlab("") +
  ylab("") +
  coord_polar(theta = "y") +
  ylim(0,max(df_suppliers$X2019+1000)) +
  theme_bw() +
  theme_opts

girafe(ggobj = v1, width_svg = 1280/72, height_svg = 1280/72,
       options = list(opts_sizing(rescale = TRUE, width = 0.50))
)
