Accessing the database

via db_connection

db <- dbDriver("SQLite")
conn <- dbConnect(db,'archetypes/world-pm25/world-pm25.db')

df <- as.data.frame(dbListTables(conn))
df

Explore

the pm25 table

df_table <- dbReadTable(conn, 'pm25')
df_table

Close

the database connection.

dbDisconnect(conn)
