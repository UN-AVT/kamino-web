url_root <- "https://raw.githubusercontent.com/UN-AVT/kamino-source/main/sources/0-shared/data/"
url_file <- "population-by-gender/gender-uganda.csv"
url <- paste0(url_root, url_file)

df <- read.csv(url, header = TRUE)
df

Plot

lines with a ribbon

theme_opts <- theme(
    text = element_text(family = "inconsolata"), 
    plot.title = element_text(color = "black", size = 18, face = "bold", family = "inconsolata"),
    plot.subtitle = element_text(color = "black", size = 16, family = "inconsolata"),
    plot.caption = element_text(color = "#555555", size = 8, family = "inconsolata"),
    panel.border = element_blank(),
    panel.background = element_blank(),
    #panel.grid.minor = element_blank(), # remove minor gridlines
    panel.grid.major.x = element_blank(), # remove x (vertical) gridlines
    legend.title = element_blank(), # remove legend title
    legend.text = element_text(color = "black", size = 8),
    legend.position='top'
)

v1 <- ggplot(df)+
  geom_line(aes(x=Year, y= M_Pop), colour= "#2da9d9", size = 1) +
  geom_line(aes(x=Year, y= F_Pop), colour= "#b73377", size = 1) +
  geom_ribbon(aes(x=Year, ymin= F_Pop, ymax = M_Pop), alpha = 0.3, fill = "#ff99cd") +
   annotate("text", x = 2010, y = 15500000, label = "Male Population", angle = 25, size = 4, colour= "#2da9d9") +
   annotate("text", x = 2006, y = 15000000, label = "Female Population", angle = 25, size = 4, colour= "#b73377") +
  labs(title = "Population",
    subtitle = "by gender",
    y = "Population",
    x = "Year",
    caption = "Data Source: World Bank") +
  theme_bw() + 
  theme_opts

girafe(ggobj = v1, width_svg = 16, height_svg = 9,
       options = list(opts_sizing(rescale = TRUE, width = 0.9)))
