Geo Circle Packing is a visualization where area of circles represents a numerical data value and within each circle, a representation of the associated geographical geometry.

Base map

natural earth

# Get countries, simplified resolution
ctry <- ne_countries(scale = "small", returnclass = "sf")

ctry <- st_transform(ctry, 3857)

# Only keep the largest polygons of multipart polygons for a few countries
frag_ctry <- c("CH", "FR", "GB", "US", "IE", "CA", "JP")
largest_ring = function(x) {
  x$ids <- 1:nrow(x)
  pols = st_cast(x, "POLYGON", warn = FALSE)
  spl = split(x = pols, f = pols$ids)
  do.call(rbind, (lapply(spl, function(y) y[which.max(st_area(y)),])))
}

st_geometry(ctry[ctry$iso_a2 %in% frag_ctry,]) <-
st_geometry(largest_ring(ctry[ctry$iso_a2 %in% frag_ctry,]))

Wrangle

merge data and circle geometry

# merge data
ctry_med <- merge(ctry[,"name_sort"], df_3, by.x = "name_sort", by.y = "exporter" )
ctry_med
# Computes circles and polygons
res_med <- popcircle(x = ctry_med, var = "exports_usd")
circles_med <- res_med$circles
circles_med$lab <- paste0(circles_med$name_sort, '\n', round(circles_med$exports_usd/100000000))
circles_med_sf = st_as_sf(circles_med)

shapes_med <- res_med$shapes
shapes_med_sf = st_as_sf(shapes_med)
