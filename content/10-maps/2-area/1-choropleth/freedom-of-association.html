Ingest

with filter by indicator

# Data source must have an ISO3 field
df <- read.csv("archetypes/freedom-of-association/freedom-of-association.csv", header = TRUE)
# Indicator: Freedom of association and assembly
df <- filter(df, Indicator_Id == 41920)
df # after filter

World map

from natural earth

# natural earth base map
ne_world <- ne_countries(scale = "small", returnclass = "sf")
# removes Antarctica
ne_world <- filter(ne_world, iso_a3 != "ATA") 
# combine data with map based on ISO3 codes
ne_world_df <- merge(ne_world, df, by.x = "iso_a3", by.y = "ISO3", all = TRUE)

Plot

Freedom of association and assembly

# theme parameters
theme_opts <- theme(
  text = element_text(family = "inconsolata"), 
  plot.title = element_text(color = "black", size = 14, face = "bold"),
  plot.subtitle = element_text(color = "black", size = 12),
  plot.caption = element_text(color = "#555555", size = 8),
  plot.background = element_blank(),
  legend.position = "bottom",
  legend.title = element_blank(),
  axis.text = element_blank(),
  axis.line = element_blank(),
  axis.ticks = element_blank(),
  axis.title = element_blank(),
  panel.grid = element_blank(),
  panel.grid.major = element_blank(),
  panel.grid.minor = element_blank(),
  panel.background=element_rect(fill="white", colour="white"),
  panel.border = element_blank()
)

# color palette to use
pal <- wes_palette("Zissou1", 10, type = "continuous")

v1 <- ggplot(data = ne_world_df) +
  # map feature styling
  geom_sf(aes(fill=X2018), color="#FFFFFF", stroke=0.5) +
  scale_fill_gradientn(colours = rev(pal)) + 
  coord_sf() +
  theme_bw() +
  labs(x="", # Longitude
       y="", # Latitude
       title = "Choropleth Map", 
       subtitle="Freedom of association and assembly") +
  theme_opts

girafe(ggobj = v1, width_svg = 16, height_svg = 7.5,
       options = list(opts_sizing(rescale = TRUE, width = 1.0))
)
