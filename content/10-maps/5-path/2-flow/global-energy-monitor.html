# Source for choropleth
df_countries <- rbind(df_sources, df_targets)
df_countries$color <- 1
df_countries$iso <- countrycode(df_countries$iso, origin = 'iso2c', destination = 'iso3c') 
df_countries <- df_countries[!duplicated(df_countries$iso), ]
